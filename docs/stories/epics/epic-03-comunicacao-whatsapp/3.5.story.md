### Helpers — schedule-utils.ts

```typescript
// Retorna true se o schedule deve disparar na hora `now` (UTC).
export function scheduleShouldFireNow(
  schedule: { frequency: string; send_day: number | null; send_time: string },
  now: Date
): boolean {
  const [sendHour] = schedule.send_time.split(':').map(Number)
  if (now.getUTCHours() !== sendHour) return false

  switch (schedule.frequency) {
    case 'daily': return true
    case 'weekly':  return now.getUTCDay() === schedule.send_day
    case 'biweekly': {
      if (now.getUTCDay() !== schedule.send_day) return false
      // Dispara em semanas pares (ISO week, modulo 2)
      const startOfYear = new Date(Date.UTC(now.getUTCFullYear(), 0, 1))
      const weekOfYear = Math.floor((now.getTime() - startOfYear.getTime()) / (7 * 86400 * 1000))
      return weekOfYear % 2 === 0
    }
    case 'monthly': return now.getUTCDate() === schedule.send_day
    default: return false
  }
}

// Retorna period_start e period_end (datas ISO YYYY-MM-DD) para o período vigente.
export function computePeriod(frequency: string, now: Date): { period_start: string; period_end: string } {
  const today = now.toISOString().split('T')[0]

  switch (frequency) {
    case 'daily': return { period_start: today, period_end: today }

    case 'weekly': {
      const day = now.getUTCDay()
      const offset = day === 0 ? 6 : day - 1  // dias desde segunda-feira
      const monday = new Date(now)
      monday.setUTCDate(monday.getUTCDate() - offset)
      return { period_start: monday.toISOString().split('T')[0], period_end: today }
    }

    case 'biweekly': {
      const day = now.getUTCDay()
      const offset = day === 0 ? 6 : day - 1
      const monday = new Date(now)
      monday.setUTCDate(monday.getUTCDate() - offset)
      const startOfYear = new Date(Date.UTC(now.getUTCFullYear(), 0, 1))
      const weekOfYear = Math.floor((monday.getTime() - startOfYear.getTime()) / (7 * 86400 * 1000))
      const blockStart = new Date(monday)
      if (weekOfYear % 2 !== 0) blockStart.setUTCDate(blockStart.getUTCDate() - 7)
      return { period_start: blockStart.toISOString().split('T')[0], period_end: today }
    }

    case 'monthly': {
      const monthStart = new Date(Date.UTC(now.getUTCFullYear(), now.getUTCMonth(), 1))
      return { period_start: monthStart.toISOString().split('T')[0], period_end: today }
    }

    default: throw new Error(`Frequência desconhecida: ${frequency}`)
  }
}
```

### Cron Route — send-reports (`apps/web/src/app/api/cron/send-reports/route.ts`)

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { createSupabaseServiceClient } from '@/lib/supabase/server'
import { generateReport } from '@/lib/whatsapp/report-generator'
import { scheduleShouldFireNow, computePeriod } from '@/lib/reports/schedule-utils'

export async function GET(req: NextRequest) {
  if (req.headers.get('Authorization') !== `Bearer ${process.env.CRON_SECRET}`) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  const supabase = createSupabaseServiceClient()
  const now = new Date()

  // 1. Buscar schedules ativos
  // NOTA: sem embed whatsapp_accounts!inner(...) — não existe FK entre
  // report_schedules e whatsapp_accounts (PostgREST exige FK para embed).
  // Cruzamento é feito no código por workspace_id (passo 2).
  const { data: schedules } = await supabase
    .from('report_schedules')
    .select('id, workspace_id, client_id, frequency, send_day, send_time, destination_type, destination_id')
    .eq('is_active', true)

  if (!schedules?.length) return NextResponse.json({ processed: 0 })

  // 2. Buscar contas WhatsApp conectadas dos workspaces envolvidos e indexar
  const workspaceIds = [...new Set(schedules.map((s) => s.workspace_id))]
  const { data: waAccounts } = await supabase
    .from('whatsapp_accounts')
    .select('workspace_id, account_id')
    .in('workspace_id', workspaceIds)
    .eq('status', 'connected')

  // 1ª conta conectada por workspace (multi-conta com preferência é trabalho futuro)
  const waByWorkspace = new Map<string, { account_id: string }>()
  for (const wa of waAccounts ?? []) {
    if (!waByWorkspace.has(wa.workspace_id)) waByWorkspace.set(wa.workspace_id, wa)
  }

  let sent = 0, skipped = 0, failed = 0

  for (const s of schedules) {
    if (!scheduleShouldFireNow(s, now)) { skipped++; continue }

    const waAccount = waByWorkspace.get(s.workspace_id)
    if (!waAccount) { skipped++; continue }  // workspace sem conta conectada

    const { period_start, period_end } = computePeriod(s.frequency, now)

    // 3. Dedup atômico — INSERT ON CONFLICT DO NOTHING
    const { data: log } = await supabase
      .from('report_logs')
      .insert({
        workspace_id: s.workspace_id,
        client_id: s.client_id,
        schedule_id: s.id,
        period_start,
        period_end,
        destination_type: s.destination_type,
        destination_id: s.destination_id,
        status: 'pending',
      })
      .select('id')
      .single()

    if (!log) { skipped++; continue }  // conflito — já enviado neste período

    // 4. Gerar texto do relatório
    const from = new Date(period_start)
    const to = new Date(period_end)
    const text = await generateReport(s.workspace_id, s.client_id, { from, to })

    // 5. Enviar via worker
    const workerRes = await fetch(`${process.env.WHATSAPP_WORKER_URL}/send`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        workspace_id: s.workspace_id,
        account_id: waAccount.account_id,
        to: s.destination_id,
        text,
      }),
    }).catch(() => null)

    if (workerRes?.ok) {
      await supabase.from('report_logs').update({ status: 'sent', sent_at: new Date().toISOString() }).eq('id', log.id)
      sent++
    } else {
      const err = workerRes ? (await workerRes.text().catch(() => 'erro desconhecido')) : 'worker inacessível'
      await supabase.from('report_logs').update({ status: 'failed', error_message: err }).eq('id', log.id)
      failed++
    }
  }

  return NextResponse.json({ processed: schedules.length, sent, skipped, failed })
}
```

**Nota:** `generateReport` retorna `string`. Se a função retornar objeto `{ text, error }`, adaptar na chamada.
