## Dev Notes

### Cálculo de ranges semanais

```typescript
// apps/web/src/lib/analytics/weekly.ts
export interface WeekRange {
  label: string  // 'S-4', 'S-3', 'S-2', 'S-1'
  start: string  // 'YYYY-MM-DD'
  end: string    // 'YYYY-MM-DD'
}

export function getWeekRanges(n = 4): WeekRange[] {
  const ranges: WeekRange[] = []
  const now = new Date()
  // Última segunda-feira passada
  const lastMonday = new Date(now)
  lastMonday.setDate(now.getDate() - now.getDay() - 6)
  lastMonday.setHours(0, 0, 0, 0)

  for (let i = n - 1; i >= 0; i--) {
    const start = new Date(lastMonday)
    start.setDate(lastMonday.getDate() - i * 7)
    const end = new Date(start)
    end.setDate(start.getDate() + 6)

    ranges.push({
      label: `S-${i + 1}`,  // S-4 é a mais antiga, S-1 a mais recente
      start: start.toISOString().split('T')[0],
      end: end.toISOString().split('T')[0],
    })
  }
  return ranges  // ordem cronológica: [S-4, S-3, S-2, S-1]
}
```

### Delta percentual

```typescript
export function percentDelta(prev: number, curr: number): number | null {
  if (prev === 0) return null  // sem dado anterior — exibir '—' na UI
  return Math.round(((curr - prev) / prev) * 100)
}
```
