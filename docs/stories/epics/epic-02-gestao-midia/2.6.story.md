## Dev Notes

### calculateHealthScore — lógica proposta

```typescript
// packages/utils/src/health.ts
export interface ClientMetrics {
  totalSpend: number
  totalRevenue: number
  totalClicks: number
  totalImpressions: number
  totalConversions: number
}

export function calculateHealthScore(metrics: ClientMetrics): number {
  if (metrics.totalSpend === 0) return 0

  const roas = metrics.totalRevenue / metrics.totalSpend
  const ctr = metrics.totalImpressions > 0
    ? metrics.totalClicks / metrics.totalImpressions
    : 0

  // ROAS: peso 60% (0-5+ mapeado para 0-60)
  const roasScore = Math.min(roas / 5, 1) * 60

  // CTR: peso 40% (0-5% mapeado para 0-40)
  const ctrScore = Math.min(ctr / 0.05, 1) * 40

  return Math.round(roasScore + ctrScore)
}
```
